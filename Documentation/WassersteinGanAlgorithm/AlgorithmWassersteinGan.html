170    def gradient_penalty(self, batch_size, real_feature, real_label, synthetic_feature):
171        """
172        Compute the gradient penalty for the Wasserstein GAN.
173
174        The gradient penalty is used to enforce the Lipschitz constraint on the discriminator's output.
175
176        Parameters:
177            batch_size (int):
178                The batch size of the input data.
179            real_feature (tensorflow.Tensor):
180                Real data features.
181            synthetic_feature (tensorflow.Tensor):
182                Synthetic (generated) data features.
183
184        """
185        # Generate random noise for smoothing.
186        random_smooth = tensorflow.random.normal([batch_size, 1], 0.0, 0.1)
187
188        # Calculate the linear distance between real and synthetic features.
189        linear_distance = synthetic_feature - real_feature
190
191        # Interpolate between real and synthetic features using the random noise.
192        interpolated_feature = real_feature + random_smooth * linear_distance
193
194        with tensorflow.GradientTape() as gradient_penalty:
195            # Watch the interpolated features for gradient computation.
196            gradient_penalty.watch(interpolated_feature)
197
198            # Get discriminator's output for the interpolated features.
199            labels_predicted = self.discriminator([interpolated_feature, real_label], training=True)
200
201        # Calculate the gradient of the discriminator's output with respect to the interpolated features.
202        gradient_computed = gradient_penalty.gradient(labels_predicted, [interpolated_feature])[0]
203
204        # Compute the gradient magnitude and normalize it.
205        gradient_normalized = tensorflow.sqrt(tensorflow.reduce_sum(tensorflow.square(gradient_computed), axis=[1]))
206
207        # Calculate the final gradient penalty as the mean squared difference from 1.0 and return.
208        gradient_penalty_final = tensorflow.reduce_mean((gradient_normalized - 1.0) ** 2)
209
210        return gradient_penalty_final

212    @tensorflow.function
213    def train_step(self, batch):
214        """
215        Executes one training step for the GAN model.
216
217        This step updates both the discriminator and the generator.
218        The discriminator is updated multiple times (controlled by self._discriminator_steps),
219        while the generator is updated once.
220
221        Args:
222            batch (tuple): A tuple containing:
223                - real_feature: A batch of real data samples (features).
224                - real_samples_label: Corresponding class labels for each sample.
225
226        Returns:
227            dict: Dictionary containing the discriminator and generator loss for the current training step.
228        """
229
230        # Unpack batch into features and labels.
231        real_feature, real_samples_label = batch
232        batch_size = tensorflow.shape(real_feature)[0]
233
234        # Expand label dimensions to match input expectations (e.g., (batch_size, 1)).
235        real_samples_label = tensorflow.expand_dims(real_samples_label, axis=-1)
236
237        # === Discriminator Training Loop ===
238        for _ in range(self._discriminator_steps):
239            # Generate random noise vectors for the latent space.
240            latent_space = tensorflow.random.normal((batch_size, self._latent_dimension),
241                                                    mean=self._latent_mean_distribution,
242                                                    stddev=self._latent_stander_deviation)
243
244            with tensorflow.GradientTape() as discriminator_gradient:
245                # Generate synthetic samples from the generator using noise and labels.
246                synthetic_feature = self._generator([latent_space, real_samples_label], training=False)
247
248                # Predict "real/fake" labels using the discriminator for real and synthetic samples.
249                label_predicted_real = self._discriminator([real_feature, real_samples_label], training=True)
250                label_predicted_synthetic = self._discriminator([synthetic_feature, real_samples_label], training=True)
251
252                # Compute discriminator loss (real vs fake).
253                discriminator_loss_result = self._discriminator_loss_fn(
254                    real_img=label_predicted_real, fake_img=label_predicted_synthetic)
255
256                # Compute gradient penalty for improved stability (WGAN-GP, etc.).
257                gradient_penalty = self.gradient_penalty(batch_size,
258                                                         real_feature,
259                                                         real_samples_label,
260                                                         synthetic_feature)
261
262                # Combine loss with gradient penalty.
263                all_discriminator_loss = discriminator_loss_result + gradient_penalty * self._gradient_penalty_weight
264
265            # Compute and apply gradients to update the discriminator's weights.
266            gradient_computed = discriminator_gradient.gradient(all_discriminator_loss,
267                                                                self.discriminator.trainable_variables)
268            self._discriminator_optimizer.apply_gradients(zip(gradient_computed,
269                                                              self.discriminator.trainable_variables))
270
271        # === Generator Training Step ===
272        # Generate fresh random noise vectors for the latent space.
273        latent_space = tensorflow.random.normal((batch_size, self._latent_dimension),
274                                                mean=self._latent_mean_distribution,
275                                                stddev=self._latent_stander_deviation)
276
277        with tensorflow.GradientTape() as generator_gradient:
278            # Generate synthetic samples from the generator.
279            synthetic_feature = self._generator([latent_space, real_samples_label], training=True)
280
281            # Predict "real/fake" labels for synthetic samples using the discriminator.
282            predicted_labels = self._discriminator([synthetic_feature, real_samples_label], training=False)
283
284            # Compute generator loss (how well generator fools the discriminator).
285            all_generator_loss = self._generator_loss_fn(predicted_labels)
286
287        # Compute and apply gradients to update the generator's weights.
288        gradient_computed = generator_gradient.gradient(all_generator_loss, self.generator.trainable_variables)
289        self._generator_optimizer.apply_gradients(zip(gradient_computed, self.generator.trainable_variables))
290
291        # Return the loss values for monitoring/tracking.
292        return {"d_loss": all_discriminator_loss, "g_loss": all_generator_loss}

294    def get_samples(self, number_samples_per_class):
295        """
296        Generates synthetic samples for each specified class using the trained generator.
297
298        This method creates samples conditioned on class labels, using random noise vectors
299        and the generator to produce the samples.
300
301        Args:
302            number_samples_per_class (dict): A dictionary containing:
303                - "classes" (dict): Mapping of class labels to the number of samples to generate for each class.
304                - "number_classes" (int): Total number of classes (used for one-hot encoding).
305
306        Returns:
307            dict: A dictionary where each key is a class label and the value is an array of generated samples.
308        """
309
310        # Dictionary to store generated samples for each class.
311        generated_data = {}
312
313        # Loop through each class and the desired number of samples for that class.
314        for label_class, number_instances in number_samples_per_class["classes"].items():
315            # Create one-hot encoded labels for all samples of the current class.
316            label_samples_generated = to_categorical([label_class] * number_instances,
317                                                     num_classes=number_samples_per_class["number_classes"])
318
319            # Sample random noise vectors from a normal distribution.
320            latent_noise = numpy.random.normal(loc=self._latent_mean_distribution,
321                                               scale=self._latent_stander_deviation,
322                                               size=(number_instances, self._latent_dimension))
323
324            # Generate synthetic samples using the generator.
325            generated_samples = self._generator.predict([latent_noise, label_samples_generated], verbose=0)
326
327            # Round the generated samples to integer values
328            # (if samples are intended to be binary, e.g., images with pixel values 0 or 1).
329            generated_samples = numpy.rint(generated_samples)
330
331            # Store generated samples for the current class.
332            generated_data[label_class] = generated_samples
333
334        # Return the dictionary containing generated samples for all requested classes.
335        return generated_data

213    @tensorflow.function
214    def train_step(self, batch):
215        """
216        Executes one training step for the GAN model.
217
218        This step updates both the discriminator and the generator.
219        The discriminator is updated multiple times (controlled by self._discriminator_steps),
220        while the generator is updated once.
221
222        Args:
223            batch (tuple): A tuple containing:
224                - real_feature: A batch of real data samples (features).
225                - real_samples_label: Corresponding class labels for each sample.
226
227        Returns:
228            dict: Dictionary containing the discriminator and generator loss for the current training step.
229        """
230
231        # Unpack batch into features and labels.
232        real_feature, real_samples_label = batch
233        batch_size = tensorflow.shape(real_feature)[0]
234
235        # Expand label dimensions to match input expectations (e.g., (batch_size, 1)).
236        real_samples_label = tensorflow.expand_dims(real_samples_label, axis=-1)
237
238        # === Discriminator Training Loop ===
239        for _ in range(self._discriminator_steps):
240            # Generate random noise vectors for the latent space.
241            latent_space = tensorflow.random.normal((batch_size, self._latent_dimension),
242                                                    mean=self._latent_mean_distribution,
243                                                    stddev=self._latent_stander_deviation)
244
245            with tensorflow.GradientTape() as discriminator_gradient:
246                # Generate synthetic samples from the generator using noise and labels.
247                synthetic_feature = self._generator([latent_space, real_samples_label], training=False)
248
249                # Predict "real/fake" labels using the discriminator for real and synthetic samples.
250                label_predicted_real = self._discriminator([real_feature, real_samples_label], training=True)
251                label_predicted_synthetic = self._discriminator([synthetic_feature, real_samples_label], training=True)
252
253                # Compute discriminator loss (real vs fake).
254                discriminator_loss_result = self._discriminator_loss_fn(
255                    real_img=label_predicted_real, fake_img=label_predicted_synthetic)
256
257                # Compute gradient penalty for improved stability (WGAN-GP, etc.).
258                gradient_penalty = self.gradient_penalty(batch_size,
259                                                         real_feature,
260                                                         real_samples_label,
261                                                         synthetic_feature)
262
263                # Combine loss with gradient penalty.
264                all_discriminator_loss = discriminator_loss_result + gradient_penalty * self._gradient_penalty_weight
265
266            # Compute and apply gradients to update the discriminator's weights.
267            gradient_computed = discriminator_gradient.gradient(all_discriminator_loss,
268                                                                self.discriminator.trainable_variables)
269            self._discriminator_optimizer.apply_gradients(zip(gradient_computed,
270                                                              self.discriminator.trainable_variables))
271
272        # === Generator Training Step ===
273        # Generate fresh random noise vectors for the latent space.
274        latent_space = tensorflow.random.normal((batch_size, self._latent_dimension),
275                                                mean=self._latent_mean_distribution,
276                                                stddev=self._latent_stander_deviation)
277
278        with tensorflow.GradientTape() as generator_gradient:
279            # Generate synthetic samples from the generator.
280            synthetic_feature = self._generator([latent_space, real_samples_label], training=True)
281
282            # Predict "real/fake" labels for synthetic samples using the discriminator.
283            predicted_labels = self._discriminator([synthetic_feature, real_samples_label], training=False)
284
285            # Compute generator loss (how well generator fools the discriminator).
286            all_generator_loss = self._generator_loss_fn(predicted_labels)
287
288        # Compute and apply gradients to update the generator's weights.
289        gradient_computed = generator_gradient.gradient(all_generator_loss, self.generator.trainable_variables)
290        self._generator_optimizer.apply_gradients(zip(gradient_computed, self.generator.trainable_variables))
291
292        # Return the loss values for monitoring/tracking.
293        return {"d_loss": all_discriminator_loss, "g_loss": all_generator_loss}

A Wasserstein Generative Adversarial Network (Wasserstein GAN) model.

This class represents a Wasserstein GAN consisting of a generator and discriminator model. It implements the Wasserstein loss with gradient penalty to improve the training of the discriminator and generator.

Reference: Arjovsky, M., Chintala, S., & Bottou, L. (2017). Wasserstein GAN. In Proceedings of the 34th International Conference on Machine Learning (ICML 2017) (Vol. 70, pp. 214-223). http://proceedings.mlr.press/v70/arjovsky17a.html

def gradient_penalty(self, batch_size, real_feature, real_label, synthetic_feature): 
171    def gradient_penalty(self, batch_size, real_feature, real_label, synthetic_feature):
172        """
173        Compute the gradient penalty for the Wasserstein GAN.
174
175        The gradient penalty is used to enforce the Lipschitz constraint on the discriminator's output.
176
177        Parameters:
178            batch_size (int):
179                The batch size of the input data.
180            real_feature (tensorflow.Tensor):
181                Real data features.
182            synthetic_feature (tensorflow.Tensor):
183                Synthetic (generated) data features.
184
185        """
186        # Generate random noise for smoothing.
187        random_smooth = tensorflow.random.normal([batch_size, 1], 0.0, 0.1)
188
189        # Calculate the linear distance between real and synthetic features.
190        linear_distance = synthetic_feature - real_feature
191
192        # Interpolate between real and synthetic features using the random noise.
193        interpolated_feature = real_feature + random_smooth * linear_distance
194
195        with tensorflow.GradientTape() as gradient_penalty:
196            # Watch the interpolated features for gradient computation.
197            gradient_penalty.watch(interpolated_feature)
198
199            # Get discriminator's output for the interpolated features.
200            labels_predicted = self.discriminator([interpolated_feature, real_label], training=True)
201
202        # Calculate the gradient of the discriminator's output with respect to the interpolated features.
203        gradient_computed = gradient_penalty.gradient(labels_predicted, [interpolated_feature])[0]
204
205        # Compute the gradient magnitude and normalize it.
206        gradient_normalized = tensorflow.sqrt(tensorflow.reduce_sum(tensorflow.square(gradient_computed), axis=[1]))
207
208        # Calculate the final gradient penalty as the mean squared difference from 1.0 and return.
209        gradient_penalty_final = tensorflow.reduce_mean((gradient_normalized - 1.0) ** 2)
210
211        return gradient_penalty_final
