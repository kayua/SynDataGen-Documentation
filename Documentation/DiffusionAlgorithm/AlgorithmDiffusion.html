 28    """
 29    Implements a diffusion process using UNet architectures for generating synthetic data.
 30    This model integrates an autoencoder and a diffusion network, enabling both data
 31    reconstruction and controlled generative modeling through Gaussian diffusion.
 32
 33    This class supports exponential moving average (EMA) updates for stable training,
 34    multiple training stages, and customizable hyperparameters to adapt to different tasks.
 35
 36    Attributes:
 37        @ema (float):
 38            Exponential moving average (EMA) decay rate for stabilizing training updates.
 39        @margin (float):
 40            Margin parameter used for loss computation or regularization purposes.
 41        @gdf_util:
 42            Utility object for Gaussian diffusion functions, handling noise scheduling and diffusion-related operations.
 43        @time_steps (int):
 44            Number of time steps used in the diffusion process.
 45        @train_stage (str):
 46            Defines the current training stage ('all', 'diffusion', etc.), determining whether only specific components are updated.
 47        @network (Model):
 48            Primary UNet model responsible for the diffusion process.
 49        @second_unet_model (Model):
 50            Secondary UNet model used for EMA-based weight updates to enhance training stability.
 51        @embedding_dimension (int):
 52            Dimensionality of the latent space used for encoding data.
 53        @encoder_model_data (Model):
 54            Encoder model responsible for feature extraction from input data.
 55        @decoder_model_data (Model):
 56            Decoder model used to reconstruct data from encoded representations.
 57        @optimizer_diffusion (Optimizer):
 58            Optimizer used for training the diffusion model.
 59        @optimizer_autoencoder (Optimizer):
 60            Optimizer responsible for training the autoencoder components.
 61        @ensemble_encoder_decoder (Model):
 62            Combined encoder-decoder model for data reconstruction.
 63
 64    Raises:
 65        ValueError:
 66            Raised in cases where:
 67            - The number of time steps is non-positive.
 68            - The EMA decay rate is outside the range (0,1).
 69            - The embedding dimension is invalid (<=0).
 70
 71    References:
 72        - Ho, J., Jain, A., & Abbeel, P. (2020). "Denoising Diffusion Probabilistic Models."
 73        Advances in Neural Information Processing Systems (NeurIPS).
 74        Available at: https://arxiv.org/abs/2006.11239

 76    Example:
 77        >>> diffusion_model = DiffusionModel(
 78        ...     first_unet_model=primary_unet,
 79        ...     second_unet_model=ema_unet,
 80        ...     encoder_model_image=encoder,

 82        ...     gdf_util=gaussian_diffusion,
 83        ...     optimizer_autoencoder=tf.keras.optimizers.Adam(learning_rate=1e-4),
 84        ...     optimizer_diffusion=tf.keras.optimizers.Adam(learning_rate=2e-4),
 85        ...     time_steps=1000,
 86        ...     ema=0.999,
 87        ...     margin=0.1,
 88        ...     embedding_dimension=128,

 90        ... )
 91        >>> diffusion_model.set_stage_training('diffusion')
 92        >>> diffusion_model.train_step(data)

189    def train_diffusion_model(self, data, ground_truth):
190        """
191        Performs a single training step for the diffusion model.
192
193        This method applies the forward diffusion process (adding noise to the data),
194        predicts the noise using the model, computes the loss, and updates the model weights.
195
196        Args:
197            data (tf.Tensor): Input data embeddings (e.g., image or text embeddings).
198            ground_truth (tf.Tensor): Corresponding class labels or conditioning embeddings.
199
200        Returns:
201            tf.Tensor: The computed loss for this training step.
202        """
203
204        # Labels (conditioning information) and input data embeddings
205        embedding_label = ground_truth
206        embedding_data_expanded = data
207
208        # Batch size of the current data batch
209        batch_size = tensorflow.shape(data)[0]
210
211        # Sample random time steps for each sample in the batch (each sample can be at a different step t)
212        random_time_steps = tensorflow.random.uniform(minval=0,
213                                                      maxval=self._time_steps,
214                                                      shape=(batch_size,),
215                                                      dtype=tensorflow.int32)
216        loss_diffusion = 0
217        # Track gradients for the diffusion model's weights
218        with tensorflow.GradientTape() as tape:
219
220            # Sample random noise to add to the data (same shape as the data itself)
221            random_noise = tensorflow.random.normal(shape=tensorflow.shape(embedding_data_expanded),
222                                                    dtype=embedding_data_expanded.dtype)
223
224            # Apply forward diffusion process (add noise based on the current time step t)
225            embedding_with_noise = self._gdf_util.q_sample(embedding_data_expanded,
226                                                           random_time_steps,
227                                                           random_noise)
228
229            # Predict noise using the diffusion model (network), conditioned on time and label
230            predicted_noise = self._network([embedding_with_noise, random_time_steps, embedding_label], training=True)
231
232            # Compute the loss by comparing the true noise with the predicted noise
233            loss_diffusion = self.loss(random_noise, predicted_noise)
234
235        # Compute gradients for the model's trainable weights
236        gradients = tape.gradient(loss_diffusion, self._network.trainable_weights)
237
238        # Apply gradients using the diffusion model's optimizer
239        self._optimizer_diffusion.apply_gradients(zip(gradients, self._network.trainable_weights))
240
241        # Return the computed diffusion loss for monitoring
242        return loss_diffusion

244    def update_ema_weights(self):
245        """
246        Updates the weights of the second UNet model using exponential moving average.
247        """
248        for weight, ema_weight in zip(self._network.weights, self._second_unet_model.weights):
249            ema_weight.assign(self._ema * ema_weight + (1 - self._ema) * weight)

251    def generate_data(self, labels, batch_size):
252        """
253        Generates synthetic data by reversing the diffusion process, starting from pure noise
254        and iteratively denoising to create data samples conditioned on class labels.
255
256        Args:
257            labels (tf.Tensor): Class labels used to condition the generated data (e.g., class embeddings).
258            batch_size (int): Number of data samples to generate in a single batch.
259
260        Returns:
261            numpy.ndarray: Generated synthetic data samples after reversing the diffusion process.
262        """
263
264        # Start with random noise in the embedding space
265        embedding_diffusion = tensorflow.random.normal(
266            shape=(labels.shape[0], self._embedding_dimension, 1),  # Shape of the noise tensor
267            dtype=tensorflow.float32
268        )
269
270        # Reshape labels to ensure they have the correct shape for conditioning
271        labels_vector = tensorflow.expand_dims(labels, axis=-1)
272
273        # Reverse the diffusion process by iterating over the time steps (from T to 0)
274        for time_step in reversed(range(0, self._time_steps)):
275
276            # Create an array with the current time step for each sample in the batch
277            array_time = tensorflow.cast(tensorflow.fill([labels_vector.shape[0]], time_step),
278                                         dtype=tensorflow.int32)
279
280            # Predict the noise at the current time step using the trained network
281            predicted_noise = self._network.predict([embedding_diffusion, array_time, labels_vector],
282                                                    verbose=0, batch_size=batch_size)
283
284            # Apply the reverse diffusion step to remove noise from the embeddings
285            embedding_diffusion = self._gdf_util.p_sample(predicted_noise[0], embedding_diffusion, array_time,
286                                                          clip_denoised=True)
287
288        # Use the decoder model to transform the denoised embeddings into real data samples
289        generated_data = self._decoder_model_data.predict([embedding_diffusion, labels_vector], verbose=0, batch_size=batch_size // 4)
290
291        # Return the generated data
292        return generated_data

294    def get_samples(self, number_samples_per_class):
295        """
296        Generates synthetic data samples for each class, using the specified number of samples per class.
297
298        Args:
299            number_samples_per_class (dict): A dictionary where the "classes" key maps to a dictionary of class labels
300                                             and their corresponding sample counts, and the "number_classes" key specifies
301                                             the total number of classes.
302
303        Returns:
304            dict: A dictionary where keys are class labels and values are the generated samples for each class.
305        """
306        generated_data = {}
307
308        # Iterate over each class and generate the specified number of samples
309        for label_class, number_instances in number_samples_per_class["classes"].items():
310            # Create one-hot encoded labels for the current class and number of instances
311            label_samples_generated = to_categorical([label_class] * number_instances,
312                                                     num_classes=number_samples_per_class["number_classes"])
313
314            # Generate synthetic data using the diffusion model (or another generation method)
315            generated_samples = self.generate_data(numpy.array(label_samples_generated, dtype=numpy.float32), batch_size=64)
316
317            # Round the generated samples to ensure valid output format (e.g., pixel values)
318            generated_samples = numpy.rint(generated_samples)
319
320            # Store the generated samples for the current class
321            generated_data[label_class] = generated_samples
322
323        return generated_data

325    def save_model(self, directory, file_name):
326        """
327        Save the encoder and decoder models in both JSON and H5 formats.
328
329        Args:
330            directory (str): Directory where models will be saved.
331            file_name (str): Base file name for saving models.
332        """

335
336        # Construct file names for encoder and decoder models

338        decoder_file_name = os.path.join(directory, f"{file_name}_decoder")
339        first_unet_file_name = os.path.join(directory, f"{file_name}_first_unet")
340        second_unet_file_name = os.path.join(directory, f"{file_name}_second_unet")
341
342        # Save encoder model
343        self._save_model_to_json(self._encoder_model_data, f"{encoder_file_name}.json")
344        self._encoder_model_data.save_weights(f"{encoder_file_name}.weights.h5")
345
346        # Save decoder model
347        self._save_model_to_json(self._decoder_model_data, f"{decoder_file_name}.json")
348        self._decoder_model_data.save_weights(f"{decoder_file_name}.weights.h5")
349
350        # Save encoder model
351        self._save_model_to_json(self._network, f"{first_unet_file_name}.json")
352        self._network.save_weights(f"{first_unet_file_name}.weights.h5")
353
354        # Save decoder model
355        self._save_model_to_json(self._second_unet_model, f"{second_unet_file_name}.json")
356        self._second_unet_model.save_weights(f"{second_unet_file_name}.weights.h5")

def train_diffusion_model(self, data, ground_truth): 
190    def train_diffusion_model(self, data, ground_truth):
191        """
192        Performs a single training step for the diffusion model.
193
194        This method applies the forward diffusion process (adding noise to the data),
195        predicts the noise using the model, computes the loss, and updates the model weights.
196
197        Args:
198            data (tf.Tensor): Input data embeddings (e.g., image or text embeddings).
199            ground_truth (tf.Tensor): Corresponding class labels or conditioning embeddings.
200
201        Returns:
202            tf.Tensor: The computed loss for this training step.
203        """
204
205        # Labels (conditioning information) and input data embeddings
206        embedding_label = ground_truth
207        embedding_data_expanded = data
208
209        # Batch size of the current data batch
210        batch_size = tensorflow.shape(data)[0]
211
212        # Sample random time steps for each sample in the batch (each sample can be at a different step t)
213        random_time_steps = tensorflow.random.uniform(minval=0,
214                                                      maxval=self._time_steps,
215                                                      shape=(batch_size,),
216                                                      dtype=tensorflow.int32)
217        loss_diffusion = 0
218        # Track gradients for the diffusion model's weights
219        with tensorflow.GradientTape() as tape:
220
221            # Sample random noise to add to the data (same shape as the data itself)
222            random_noise = tensorflow.random.normal(shape=tensorflow.shape(embedding_data_expanded),
223                                                    dtype=embedding_data_expanded.dtype)
224
225            # Apply forward diffusion process (add noise based on the current time step t)
226            embedding_with_noise = self._gdf_util.q_sample(embedding_data_expanded,
227                                                           random_time_steps,
228                                                           random_noise)
229
230            # Predict noise using the diffusion model (network), conditioned on time and label
231            predicted_noise = self._network([embedding_with_noise, random_time_steps, embedding_label], training=True)
232
233            # Compute the loss by comparing the true noise with the predicted noise
234            loss_diffusion = self.loss(random_noise, predicted_noise)
235
236        # Compute gradients for the model's trainable weights
237        gradients = tape.gradient(loss_diffusion, self._network.trainable_weights)
238
239        # Apply gradients using the diffusion model's optimizer
240        self._optimizer_diffusion.apply_gradients(zip(gradients, self._network.trainable_weights))
241
242        # Return the computed diffusion loss for monitoring
243        return loss_diffusion

253        """
254        Generates synthetic data by reversing the diffusion process, starting from pure noise
255        and iteratively denoising to create data samples conditioned on class labels.
256
257        Args:
258            labels (tf.Tensor): Class labels used to condition the generated data (e.g., class embeddings).
259            batch_size (int): Number of data samples to generate in a single batch.
260
261        Returns:
262            numpy.ndarray: Generated synthetic data samples after reversing the diffusion process.
263        """
264
265        # Start with random noise in the embedding space
266        embedding_diffusion = tensorflow.random.normal(
267            shape=(labels.shape[0], self._embedding_dimension, 1),  # Shape of the noise tensor
268            dtype=tensorflow.float32
269        )
270
271        # Reshape labels to ensure they have the correct shape for conditioning
272        labels_vector = tensorflow.expand_dims(labels, axis=-1)
273
274        # Reverse the diffusion process by iterating over the time steps (from T to 0)
275        for time_step in reversed(range(0, self._time_steps)):
276
277            # Create an array with the current time step for each sample in the batch

283                                                    verbose=0, batch_size=batch_size)
284
285            # Apply the reverse diffusion step to remove noise from the embeddings

287                                                          clip_denoised=True)
288
289        # Use the decoder model to transform the denoised embeddings into real data samples
290        generated_data = self._decoder_model_data.predict([embedding_diffusion, labels_vector], verbose=0, batch_size=batch_size // 4)

292        # Return the generated data
