 35class VariationalAlgorithm(Model):
 36    """
 37    Implements a Variational AutoEncoder (VAE) model for generating synthetic data.
 38
 39    The model includes an encoder and a decoder for encoding input data and reconstructing
 40    it from a learned latent space. During training, it computes both the reconstruction loss
 41    and the KL divergence loss. The trained decoder can be used to generate synthetic data.
 42
 43    This class supports customizable latent space parameters and loss functions, making it
 44    adaptable for different generative tasks.
 45
 46    Attributes:
 47        @_encoder (Model):
 48            Encoder model that encodes input data into the latent space.
 49        @_decoder (Model):
 50            Decoder model that reconstructs data from the latent representation.
 51        @_loss_function (callable):
 52            Function used to compute the total loss during training.
 53        @_total_loss_tracker (Mean):
 54            Tracks the overall loss during training.
 55        @_reconstruction_loss_tracker (Mean):
 56            Tracks the reconstruction loss during training.
 57        @_kl_loss_tracker (Mean):
 58            Tracks the KL divergence loss during training.
 59        @_latent_mean_distribution (float):
 60            Mean of the latent distribution.
 61        @_latent_stander_deviation (float):
 62            Standard deviation of the latent distribution.
 63        @_latent_dimension (int):
 64            Dimensionality of the latent space.
 65        @_decoder_latent_dimension (int):
 66            Dimensionality of the latent space used by the decoder.
 67        @_file_name_encoder (str):
 68            File name for saving the encoder model.
 69        @_file_name_decoder (str):
 70            File name for saving the decoder model.
 71        @_models_saved_path (str):
 72            Directory path where the encoder and decoder models are saved.
 73
 74    Raises:
 75        ValueError:
 76            Raised in cases where:
 77            - The latent dimension is non-positive.
 78            - The standard deviation of the latent space is non-positive.
 79            - The file paths are invalid.
 80
 81    Example:
 82        >>> vae_model = VariationalAlgorithm(
 83        ...     encoder_model=encoder,
 84        ...     decoder_model=decoder,
 85        ...     loss_function=custom_loss_function,
 86        ...     latent_dimension=128,
 87        ...     latent_mean_distribution=0.0,
 88        ...     latent_stander_deviation=1.0,
 89        ...     file_name_encoder="encoder_model.h5",
 90        ...     file_name_decoder="decoder_model.h5",
 91        ...     models_saved_path="models/"
 92        ... )
 93        >>> vae_model.train_step(data)

167    @tensorflow.function
168    def train_step(self, batch):
169        """
170        Perform a training step for the Variational AutoEncoder (VAE).
171
172        Args:
173            batch: Input data batch.
174
175        Returns:
176            dict: Dictionary containing the loss values (total loss, reconstruction loss, KL divergence loss).
177        """
178        # Use tf.function decorator for improved TensorFlow performance
179        batch_x, batch_y = batch
180
181        with tensorflow.GradientTape() as tape:
182            # Forward pass: Encode input data and sample from the latent space
183            latent_mean, latent_log_variation, latent, label = self._encoder(batch_x)
184
185            # Decode the sampled latent space and generate reconstructed data
186            reconstruction_data = self._decoder([latent, label])
187
188            # Calculate binary cross-entropy loss for reconstruction
189            binary_cross_entropy_loss = tensorflow.keras.losses.binary_crossentropy(batch_y, reconstruction_data)
190            sum_reduced = binary_cross_entropy_loss
191            reconstruction_loss = tensorflow.reduce_mean(sum_reduced)
192
193            # Calculate KL divergence loss
194            encoder_output = (1 + latent_log_variation - tensorflow.square(latent_mean))
195            kl_divergence_loss = -0.5 * (encoder_output - tensorflow.exp(latent_log_variation))
196            kl_divergence_loss = tensorflow.reduce_mean(tensorflow.reduce_sum(kl_divergence_loss, axis=1))
197
198            # Total loss is the sum of reconstruction loss and KL divergence loss
199            loss_model_in_reconstruction = reconstruction_loss + kl_divergence_loss
200
201        # Compute gradients and update model weights
202        gradient_update = tape.gradient(loss_model_in_reconstruction, self.trainable_weights)
203
204        # Update loss metrics
205        self.optimizer.apply_gradients(zip(gradient_update, self.trainable_weights))
206        self._total_loss_tracker.update_state(loss_model_in_reconstruction)
207        self._reconstruction_loss_tracker.update_state(reconstruction_loss)
208        self._kl_loss_tracker.update_state(kl_divergence_loss)
209
210        # Return a dictionary containing the current loss values
211        return {"loss": self._total_loss_tracker.result(),
212                "reconstruction_loss": self._reconstruction_loss_tracker.result(),
213                "kl_loss": self._kl_loss_tracker.result()}

236    def get_samples(self, number_samples_per_class):
237        """
238        Generate synthetic samples for each specified class using the trained decoder.
239
240        This function generates samples by sampling from a normal distribution in the latent space
241        and conditioning the generation process on class labels.
242
243        Args:
244            number_samples_per_class (dict):
245                Dictionary specifying the number of samples to generate for each class.
246                Expected structure:
247                {
248                    "classes": {class_label: number_of_samples, ...},
249                    "number_classes": total_number_of_classes
250                }
251
252        Returns:
253            dict:
254                A dictionary where each key is a class label and the value is an array of generated samples.
255                Each array contains the synthetic samples generated for the corresponding class.
256        """
257
258        # Initialize a dictionary to store the generated samples for each class
259        generated_data = {}
260
261        # Iterate over each class and the corresponding number of samples to generate
262        for label_class, number_instances in number_samples_per_class["classes"].items():
263            # Create a one-hot encoded label array for all samples in the current class
264            # Example: if label_class = 1 and number_instances = 3, this generates:
265            # [[0, 1], [0, 1], [0, 1]]
266            label_samples_generated = to_categorical([label_class] * number_instances,
267                                                     num_classes=number_samples_per_class["number_classes"])
268
269            # Sample random latent vectors from a standard normal distribution
270            # Shape: (number_instances, decoder_latent_dimension)
271            latent_noise = numpy.random.normal(size=(number_instances, self._decoder_latent_dimension))
272
273            # Use the decoder to generate samples conditioned on the latent vectors and class labels
274            # Inputs: (latent vectors, class labels)
275            # 'verbose=0' suppresses any print output from the prediction process
276            generated_samples = self._decoder.predict([latent_noise, label_samples_generated], verbose=0)
277
278            # Round the generated samples to the nearest integer
279            # This is useful for discrete data, like binary features (0/1) or integer values
280            generated_samples = numpy.rint(generated_samples)
281
282            # Store the generated samples in the dictionary under the corresponding class label
283            generated_data[label_class] = generated_samples
284
285        # Return the dictionary with all generated samples, organized by class
286        return generated_data

289    def generate_synthetic_data(self, number_samples_generate, labels, latent_dimension):
290        """
291        Generate synthetic data using the Variational AutoEncoder (VAE).
292
293        Args:
294            number_samples_generate (int): Number of synthetic samples to generate.
295            labels: Labels for the generated data.
296            latent_dimension (int): Dimension of the latent space.
297
298        Returns:
299            tensorflow.Tensor: Synthetic data generated by the decoder.
300        """
301
302        # Generate random noise samples in the latent space
303        random_noise_generate = tensorflow.random.normal(shape=(number_samples_generate, latent_dimension),
304                                                 mean=self.latent_mean_distribution, stddev=self.latent_deviation,
305                                                 dtype=tensorflow.float32)
306
307        # Create label vectors for the generated data
308        label_list = tensorflow.cast(tensorflow.fill((number_samples_generate, 1), labels), dtype=tensorflow.float32)
309
310        # Generate synthetic data by passing random noise and labels through the decoder
311        synthetic_data = self._decoder.predict(numpy.array([random_noise_generate, label_list]))
312
313        # Return the generated synthetic data as a TensorFlow tensor
314        return synthetic_data

Implements a Variational AutoEncoder (VAE) model for generating synthetic data.

The model includes an encoder and a decoder for encoding input data and reconstructing it from a learned latent space. During training, it computes both the reconstruction loss and the KL divergence loss. The trained decoder can be used to generate synthetic data.

This class supports customizable latent space parameters and loss functions, making it adaptable for different generative tasks.

Attributes: @_encoder (Model): Encoder model that encodes input data into the latent space. @_decoder (Model): Decoder model that reconstructs data from the latent representation. @_loss_function (callable): Function used to compute the total loss during training. @_total_loss_tracker (Mean): Tracks the overall loss during training. @_reconstruction_loss_tracker (Mean): Tracks the reconstruction loss during training. @_kl_loss_tracker (Mean): Tracks the KL divergence loss during training. @_latent_mean_distribution (float): Mean of the latent distribution. @_latent_stander_deviation (float): Standard deviation of the latent distribution. @_latent_dimension (int): Dimensionality of the latent space. @_decoder_latent_dimension (int): Dimensionality of the latent space used by the decoder. @_file_name_encoder (str): File name for saving the encoder model. @_file_name_decoder (str): File name for saving the decoder model. @_models_saved_path (str): Directory path where the encoder and decoder models are saved.

Raises: ValueError: Raised in cases where: - The latent dimension is non-positive. - The standard deviation of the latent space is non-positive. - The file paths are invalid.

Example:

vae_model = VariationalAlgorithm( ... encoder_model=encoder, ... decoder_model=decoder, ... loss_function=custom_loss_function, ... latent_dimension=128, ... latent_mean_distribution=0.0, ... latent_stander_deviation=1.0, ... file_name_encoder="encoder_model.h5", ... file_name_decoder="decoder_model.h5", ... models_saved_path="models/" ... ) vae_model.train_step(data)

def train_step(self, batch): 
168    @tensorflow.function
169    def train_step(self, batch):
170        """
171        Perform a training step for the Variational AutoEncoder (VAE).
172
173        Args:
174            batch: Input data batch.
175
176        Returns:
177            dict: Dictionary containing the loss values (total loss, reconstruction loss, KL divergence loss).
178        """
179        # Use tf.function decorator for improved TensorFlow performance
180        batch_x, batch_y = batch
181
182        with tensorflow.GradientTape() as tape:
183            # Forward pass: Encode input data and sample from the latent space
184            latent_mean, latent_log_variation, latent, label = self._encoder(batch_x)
185
186            # Decode the sampled latent space and generate reconstructed data
187            reconstruction_data = self._decoder([latent, label])
188
189            # Calculate binary cross-entropy loss for reconstruction
190            binary_cross_entropy_loss = tensorflow.keras.losses.binary_crossentropy(batch_y, reconstruction_data)
191            sum_reduced = binary_cross_entropy_loss
192            reconstruction_loss = tensorflow.reduce_mean(sum_reduced)
193
194            # Calculate KL divergence loss
195            encoder_output = (1 + latent_log_variation - tensorflow.square(latent_mean))
196            kl_divergence_loss = -0.5 * (encoder_output - tensorflow.exp(latent_log_variation))
197            kl_divergence_loss = tensorflow.reduce_mean(tensorflow.reduce_sum(kl_divergence_loss, axis=1))
198
199            # Total loss is the sum of reconstruction loss and KL divergence loss
200            loss_model_in_reconstruction = reconstruction_loss + kl_divergence_loss
201
202        # Compute gradients and update model weights
203        gradient_update = tape.gradient(loss_model_in_reconstruction, self.trainable_weights)
204
205        # Update loss metrics
206        self.optimizer.apply_gradients(zip(gradient_update, self.trainable_weights))
207        self._total_loss_tracker.update_state(loss_model_in_reconstruction)
208        self._reconstruction_loss_tracker.update_state(reconstruction_loss)
209        self._kl_loss_tracker.update_state(kl_divergence_loss)
210
211        # Return a dictionary containing the current loss values
212        return {"loss": self._total_loss_tracker.result(),
213                "reconstruction_loss": self._reconstruction_loss_tracker.result(),
214                "kl_loss": self._kl_loss_tracker.result()}
