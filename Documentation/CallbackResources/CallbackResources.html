 38class ResourceMonitorCallback(Callback):
 39    """
 40    Callback for monitoring system resources during Keras model training.
 41
 42    This callback monitors resources such as CPU, GPU, RAM, storage, temperature, and network
 43    and records them in a JSON file at user-defined intervals.
 44
 45    Parameters:
 46        @json_file_path (str): Path to the JSON file where the data will be saved.
 47        @save_interval (int): Interval of epochs for saving the data. By default, it is 1 (every epoch).
 48    """
 49
 50    def __init__(self, json_file_path, k_fold, save_interval=1):
 51        super(ResourceMonitorCallback, self).__init__()
 52        self.json_file_path = '{}/monitor_resource_{}_fold.json'.format(json_file_path, k_fold)
 53        self.save_interval = save_interval
 54        self.machine_info = self._get_machine_info()
 55        self.epoch_counter = 0
 56
 57    def _get_machine_info(self):
 58        """
 59        Collects basic system information at the start of training.
 60
 61        Returns:
 62            dict: Dictionary containing platform details, Python version, TensorFlow version,
 63                  CPU, memory, disk information, and placeholders for epoch statistics.
 64        """
 65        try:
 66            machine_info = {
 67                "platform": platform.platform(),
 68                "tensorflow_version": tensorflow.__version__,
 69                "numpy_version": numpy.__version__,
 70                "python_version": platform.python_version(),
 71                "cpu_info": self._get_cpu_info(),
 72                "memory_info": self._get_memory_info(),
 73                "disk_info": self._get_disk_info(),
 74                "epoch": [],
 75                "cpu_percent": {"mean": [], "per_core": []},
 76                "ram_usage": [],
 77                "swap_usage": [],
 78                "storage_available": [],
 79                "cpu_temperature": {"mean": [], "per_core": []},
 80                "gpu_temperature": [],
 81                "network_bandwidth": [],
 82                "network_transfer_rate": []
 83            }
 84            return machine_info
 85
 86        except Exception as e:
 87            logging.error(f'Failed to collect machine info: {e}')
 88            return {}
 89
 90    @staticmethod
 91    def _get_cpu_info():
 92        """
 93        Retrieves detailed CPU specifications.
 94        """
 95        try:
 96            cpu_info = subprocess.check_output(['lscpu']).decode()
 97            logging.info('CPU info collected.')
 98            return cpu_info
 99
100        except (FileNotFoundError, subprocess.CalledProcessError) as e:
101            logging.error(f'Error collecting CPU info: {e}')
102            return "Unknown CPU info"
103

105    def _get_memory_info():
106        """
107        Retrieves memory statistics.
108        """
109        try:
110            memory_info = subprocess.check_output(['free', '-h']).decode()
111            logging.debug('Memory info collected.')
112            return memory_info
113
114        except (FileNotFoundError, subprocess.CalledProcessError) as e:
115            logging.error(f'Error collecting memory info: {e}')
116            return "Unknown memory info"
117
118    @staticmethod
119    def _get_disk_info():
120        """
121        Retrieves disk storage statistics.
122        """
123        try:
124            disk_info = subprocess.check_output(['df', '-h']).decode()
125            logging.debug('Disk info collected.')
126            return disk_info
127
128        except (FileNotFoundError, subprocess.CalledProcessError) as e:
129            logging.error(f'Error collecting disk info: {e}')
130            return "Unknown disk info"

132    def on_train_begin(self, logs=None):
133        """
134        Saves the initial machine configuration when training begins.
135        """
136        try:
137
138            with open(self.json_file_path, 'w') as json_file:
139                json.dump(self.machine_info, json_file, indent=4)
140                json_file.write('\n')
141
142            logging.info('Training started and initial machine info saved.')
143
144        except IOError as e:
145            logging.error(f'Error writing to JSON file at training start: {e}')
146
147    def on_epoch_end(self, epoch, logs=None):
148        """
149        Logs system resource usage at the end of each epoch.
150        """
151        self.epoch_counter += 1
152        logging.info(f'Epoch {epoch} ended. Logs: {logs}')
153
154        if self.epoch_counter % self.save_interval == 0:
155            try:
156                cpu_percent = psutil.cpu_percent(interval=None, percpu=True)
157                cpu_percent_mean = psutil.cpu_percent(interval=None)
158                ram_usage = psutil.virtual_memory().percent
159                swap_usage = psutil.swap_memory().percent
160                storage_available = psutil.disk_usage('/').percent
161
162                data = {
163                    "epoch": epoch,
164                    "cpu_percent": {"mean": cpu_percent_mean, "per_core": cpu_percent},
165                    "ram_usage": ram_usage,
166                    "swap_usage": swap_usage,
167                    "storage_available": storage_available,
168                }
169
170                with open(self.json_file_path, 'r+') as json_file:
171                    try:
172                        data_list = json.load(json_file)
173                    except json.JSONDecodeError as e:
174                        logging.error(f'Error loading JSON file: {e}')
175                        data_list = self.machine_info  # Initialize with machine info
176
177                    for key, value in data.items():
178                        if isinstance(value, dict):
179                            for subkey, subvalue in value.items():
180                                data_list[key][subkey].append(subvalue)
181                        else:
182                            data_list[key].append(value)
183
184                    json_file.seek(0)
185                    json.dump(data_list, json_file, indent=4)
186                    json_file.truncate()
187                logging.info(f'Resource data for epoch {epoch} saved successfully.')
188
189            except IOError as e:
190                logging.error(f'Error writing resource data for epoch {epoch} to JSON: {e}')

192    @staticmethod
193    def _get_gpu_info():
194
195        try:
196            result = subprocess.check_output(
197                ['nvidia-smi', '--query-gpu=utilization.gpu,memory.used', '--format=csv,noheader,nounits'])
198            gpu_data = result.decode('utf-8').strip().split('\n')
199            gpu_utilization = int(gpu_data[0].split(',')[0])
200            gpu_memory_usage = int(gpu_data[0].split(',')[1])
201            logging.debug('GPU info collected.')
202            return gpu_utilization, gpu_memory_usage
203
204        except (FileNotFoundError, subprocess.CalledProcessError) as e:
205
206            logging.error(f'Error collecting GPU info: {e}')
207            return None, None

211
212        try:
213
214            cpu_temp_result = subprocess.check_output(['sensors'])

216            cpu_temps = [line.split(':')[-1].strip().split()[0] for line in cpu_temp_data if 'Core' in line]
217            cpu_temperature = {

219                "per_core": [float(temp.replace('°C', '')) for temp in cpu_temps]
220            }
221            logging.debug('CPU temperature info collected.')
222
223        except (FileNotFoundError, subprocess.CalledProcessError) as e:
224            logging.error(f'Error collecting CPU temperature info: {e}')
225            cpu_temperature = {"mean": None, "per_core": None}

229            gpu_temp_result = subprocess.check_output(
230                ['nvidia-smi', '--query-gpu=temperature.gpu', '--format=csv,noheader,nounits'])
231            gpu_temperature = int(gpu_temp_result.decode('utf-8').strip())
232            logging.debug('GPU temperature info collected.')
233
234        except (FileNotFoundError, subprocess.CalledProcessError) as e:

236            gpu_temperature = None
237
238        return cpu_temperature, gpu_temperature

240    @staticmethod
241    def _get_network_info():
242
243        try:
244            network_stats = psutil.net_io_counters()
245            bandwidth = network_stats.bytes_sent + network_stats.bytes_recv
246            transfer_rate = network_stats.bytes_sent / (1024 * 1024)  # In MB
247            logging.debug('Network info collected.')
248            return bandwidth, transfer_rate
249
250        except Exception as e:
251            logging.error(f'Error collecting network info: {e}')
252            return None, None
